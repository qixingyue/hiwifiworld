get_cron_job (args)
获取定时任务

Parameters:

Returns:

    {code, msg, data}

Usage:

     ret = get_cron_job(args)

 如果要获取所有设备的所有定时任务，请不要设置这些参数
 args = {
  vendor="haier",         // 厂商编码，只获取此厂商
  iot_id="abc123",        // 设备id，只获取此设备
  job_id="xyz34455"       // 定时任务id，只获取此任务
 }

 data = {
  cron_job_list = {           // 定时任务列表
    {
      job_id="xyz344",        // 定时任务id
      iot_id="abc123",        // 设备id
      vendor="haier",         // 厂商
      action="set",           // 要执行的动作，默认set
      arguments={
        outletStatus=1|0,     // 比如海尔插座，1为开，0为关
      },                      // 要执行的动作对应的参数
      minute=0|59,            // 分
      hour=0|23,              // 时
      day=*|1|31,             // 日
      month=*|1|12,           // 月
      week=*|0|6,             // 周几
      repeatflag=0|1          // 是否重复执行，默认1重复，0不重复仅1次
    },
    {
      job_id="xyz399",        // 定时任务id
      iot_id="abc777",        // 设备id
      vendor="haier",         // 厂商
      action="set",           // 要执行的动作，默认set
      arguments={
        outletStatus=1|0,     // 比如海尔插座，1为开，0为关
      },                      // 要执行的动作对应的参数
      minute=0|59,            // 分
      hour=0|23,              // 时
      day=*|1|31,             // 日
      month=*|1|12,           // 月
      week=*|0|6,             // 周几
      repeatflag=0|1          // 是否重复执行，默认1重复，0不重复仅1次
    }
  }
 }
list (args)
获取列表

Parameters:

Returns:

    Table {app_code, app_msg, app_data}

Usage:

    
 args={
  "vendor":"haier",       // vendor为可选字段，不传入时，表示获取所有厂商
  "argument":{
    "group_id":"234567"   // group_id 为可选字段，不同传入的表现如下：
                              不传时，获取所有的iot列表;
                              传入固定某个组id时，获取属于此组id下的所有iot设备列表(仅易能智达使用);
                              传入的id为空字符串""，表示获取不属于任何组id下的所有iot设备列表;
  }
 }

 app_data={
  "list": [
    {
      "mac": "60:C5:A8:60:D5:F9",

      "vendor": "enzd",
      "seq": 0, //设备序号
      "password": "172168",
      "group_id": ""
      "src": ""
      "dst": ""
    }
  ]
 }
