list (args)
获取列表

Parameters:

  • args {}

Returns:

    Table {app_code, app_msg, app_data}

Usage:

    
 args={
  date="20160316",            // 获取某天连接过的设备，不设置则获取全部,可选项
  mac="CC:78:5F:07:D6:13",    // 获取单台设备,可选项
  conn_ap="D4:EE:07:08:35:B4", // 获取某台路由器/ap/极卫星下的设备,可选项
  master=1/0 //获取主网络or访客设备列表,可选项
 }

 app_data={
  "list": [
   {
    "mac": "F8:BC:12:9E:A6:3A",
    "ip": "192.168.199.230",
    "type": "station", //"station"/"iot"/"ap"/"star"
    "name": "wenjue-pc",
    "conn_ap": "D4:EE:07:08:35:B4",
    "conn_aptype": "star", //"ap"/"router"
    "rssi": 0, //已经计算完成的信号强度
　  "phy": "2.4G", //"2.4G"/"5G"/"wire"/"usb"/"bt"
    "master": 1, //主人网络标识，1表示主人网络，0表示访客网络
    "online": 1, //在线状态，1表示在线，0表示离线
    "last_online": 1453089011, //最近一次上线时间
    "last_offline": 0，//最后一次离线时间，0表示设备从未离线
    "first_online": 1453089011, //第一次接入时间
   }
  ]
 }
