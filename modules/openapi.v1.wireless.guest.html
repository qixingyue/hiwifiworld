Module openapi.v1.wireless.guest

访客网络管理
[ 注意：由于页面上访客网络没有区分2.4G和5G，默认访客网络2.4G和5G是合并的，因此上层调用时，只需要调用2.4G接口就可以了 ]

          work=1|0,
          show=1|0,
          start_time=1424862334,
          stop_time=1424905534,
          show_time=1424862334,

          online_count=12,  #当前连接到2.4G/5G的设备数（只包括访客网络）
          max_online_count=32,  #最大允许连接2.4G/5G的设备数
          free_count=20,  #2.4G/5G剩余允许连接数
          allow_link=o|1,  #是否允许新设备连接
          systime=1426565275,  #系统时间
          hide_timeout=0|2,  #0为不隐藏，2为2分钟后隐藏
          stop_timeout=0|720,  #0为不停用，720为720分钟后停用

 }
